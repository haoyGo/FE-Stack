``` js
// 使得compose成立
const test1 = (next) => {
  console.log("test1");
  next();
  console.log("test1 end");
};

const test2 = (next) => {
  console.log("test2");
  next();
  console.log("test2 end");
};

const test3 = (next) => {
  console.log("test3");
  next();
  console.log("test3 end");
};

// 解法1
const compose = (middlewares) => {
  return () => {
    const dispatch = (i) => {
      if (i === middlewares.length) return () => {};
      const fn = middlewares[i];
      return () => fn(dispatch(i + 1));
    };
    return dispatch(0)();
  };
};

// 解法2
const compose = (middlewares) => {
  return () => {
    const dispatch = (i = 0) => {
      if (i === middlewares.length) return () => {};
      const fn = middlewares[i];
      return fn(dispatch.bind(null, i + 1));
    };
    return dispatch();
  };
};

compose([test1, test2, test3])();
```
